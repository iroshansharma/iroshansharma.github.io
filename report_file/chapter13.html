13.3.2 Salary Calculation Engine

// Comprehensive Salary Calculation Function function calculateEmployeeSalary(employeeId, payrollPeriod) { // Get employee details and salary structure const employee = getEmployeeDetails(employeeId); const salaryStructure = getEmployeeSalaryStructure(employeeId); const attendanceData = getAttendanceData(employeeId, payrollPeriod); const payrollSettings = getPayrollSettings(); // Initialize calculation object const salaryCalculation = { employee_id: employeeId, employee_name: employee.name, designation: employee.designation, department: employee.department, pay_period: payrollPeriod, earnings: {}, deductions: {}, net_pay: 0, calculation_date: new Date() }; // Calculate Basic Earnings const workingDays = payrollSettings.working_days_per_month; const actualWorkingDays = attendanceData.present_days + attendanceData.half_days * 0.5; const lossOfPayDays = Math.max(0, workingDays - actualWorkingDays); // Basic Salary Calculation const basicSalary = salaryStructure.basic_salary; const dailyBasic = basicSalary / workingDays; const earnedBasic = basicSalary - (lossOfPayDays * dailyBasic); salaryCalculation.earnings.basic_salary = earnedBasic; // HRA Calculation const hraRate = salaryStructure.hra_percentage || 50; const hra = (earnedBasic * hraRate) / 100; salaryCalculation.earnings.hra = hra; // Other Allowances salaryCalculation.earnings.transport_allowance = salaryStructure.transport_allowance || 0; salaryCalculation.earnings.medical_allowance = salaryStructure.medical_allowance || 0; salaryCalculation.earnings.special_allowance = salaryStructure.special_allowance || 0; // Overtime Calculation if (attendanceData.overtime_hours > 0) { const hourlyRate = basicSalary / (workingDays * payrollSettings.working_hours_per_day); const overtimePay = attendanceData.overtime_hours * hourlyRate * payrollSettings.overtime_rate; salaryCalculation.earnings.overtime_pay = overtimePay; } // Calculate Gross Salary const grossSalary = Object.values(salaryCalculation.earnings) .reduce((sum, amount) => sum + amount, 0); salaryCalculation.gross_salary = grossSalary; // Calculate Statutory Deductions // PF Deduction const pfEligibleSalary = Math.min(earnedBasic, payrollSettings.pf_ceiling || 15000); const pfDeduction = (pfEligibleSalary * payrollSettings.pf_rate) / 100; salaryCalculation.deductions.pf_deduction = pfDeduction; // ESI Deduction if (grossSalary <= (payrollSettings.esi_ceiling || 21000)) { const esiDeduction = (grossSalary * payrollSettings.esi_rate) / 100; salaryCalculation.deductions.esi_deduction = esiDeduction; } // Professional Tax salaryCalculation.deductions.professional_tax = payrollSettings.professional_tax || 200; // TDS Calculation (simplified) const annualSalary = grossSalary * 12; if (annualSalary > payrollSettings.tds_threshold) { const tdsDeduction = calculateTDS(annualSalary) / 12; salaryCalculation.deductions.tds = tdsDeduction; } // Other Deductions const loanDeductions = getLoanDeductions(employeeId, payrollPeriod); const advanceDeductions = getAdvanceDeductions(employeeId, payrollPeriod); salaryCalculation.deductions.loan_deduction = loanDeductions; salaryCalculation.deductions.advance_deduction = advanceDeductions; // Late Coming Penalty if (attendanceData.late_hours > 0) { const latePenalty = attendanceData.late_hours * payrollSettings.late_deduction_per_hour; salaryCalculation.deductions.late_penalty = latePenalty; } // Calculate Total Deductions const totalDeductions = Object.values(salaryCalculation.deductions) .reduce((sum, amount) => sum + amount, 0); salaryCalculation.total_deductions = totalDeductions; // Calculate Net Pay salaryCalculation.net_pay = grossSalary - totalDeductions; // Save calculation to database savePayrollCalculation(salaryCalculation); return salaryCalculation; }

13.3.3 Tax Deduction at Source (TDS) Calculation

TDS Calculation Logic

function calculateTDS(annualSalary) { // Income Tax Slabs for FY 2024-25 (Old Regime) const taxSlabs = [ { min: 0, max: 250000, rate: 0 }, { min: 250000, max: 500000, rate: 5 }, { min: 500000, max: 1000000, rate: 20 }, { min: 1000000, max: Infinity, rate: 30 } ]; // Standard deduction const standardDeduction = 50000; const taxableIncome = Math.max(0, annualSalary - standardDeduction); let totalTax = 0; let remainingIncome = taxableIncome; for (const slab of taxSlabs) { if (remainingIncome <= 0) break; const slabIncome = Math.min(remainingIncome, slab.max - slab.min); const slabTax = (slabIncome * slab.rate) / 100; totalTax += slabTax; remainingIncome -= slabIncome; } // Add Health and Education Cess (4%) totalTax += totalTax * 0.04; return Math.round(totalTax); }

13.5 Payroll Processing Automation

The system provides automated payroll processing capabilities to handle bulk salary calculations and payments.

13.5.1 Bulk Payroll Processing

// Bulk Payroll Processing Function function processBulkPayroll(payrollPeriod, departmentFilter = null) { const processingResults = { total_employees: 0, successful_calculations: 0, failed_calculations: 0, total_gross_salary: 0, total_deductions: 0, total_net_pay: 0, errors: [], processing_time: null }; const startTime = new Date(); try { // Get list of active employees const employees = getActiveEmployees(departmentFilter); processingResults.total_employees = employees.length; for (const employee of employees) { try { const salaryData = calculateEmployeeSalary(employee.id, payrollPeriod); savePayrollCalculation(salaryData); processingResults.successful_calculations++; processingResults.total_gross_salary += salaryData.gross_salary; processingResults.total_deductions += salaryData.total_deductions; processingResults.total_net_pay += salaryData.net_pay; } catch (error) { processingResults.failed_calculations++; processingResults.errors.push({ employee_id: employee.id, error: error.message }); } } processingResults.processing_time = (new Date()) - startTime; return processingResults; } catch (error) { throw new Error('Bulk payroll processing failed: ' + error.message); } }
