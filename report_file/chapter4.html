class CRMDashboard {
    constructor() {
        this.apiBase = '/api';
        this.init();

        await this.loadDashboardData();

    updateDashboard(data) {

        document.getElementById('active-leads').textContent = activeLeads;
        document.getElementById('monthly-revenue').textContent = `$${monthlyRevenue}`;
    }

    setupEventListeners() {
        document.addEventListener('click', (e) => {
            if (e.target.matches('.btn-refresh')) {
                this.loadDashboardData();
            }
        });
    }
}

Security Implementation Example
function login($username, $password) {
    global $conn;
    
    // Input validation
    if (empty($username) || empty($password)) {
        return false;
    }
    
    try {
        // Prepared statement to prevent SQL injection
        $stmt = $conn->prepare("SELECT id, username, password_hash, role_id 
                               FROM users WHERE username = ?");
        $stmt->bind_param('s', $username);
        $stmt->execute();
        $result = $stmt->get_result();
        
        if ($result->num_rows === 1) {
            $user = $result->fetch_assoc();
            
            // Verify password using secure hashing
            if (password_verify($password, $user['password_hash'])) {
                // Regenerate session ID for security
                session_regenerate_id(true);
                
                // Set session variables
                $_SESSION['username'] = $user['username'];
                $_SESSION['user_id'] = $user['id'];
                $_SESSION['role_id'] = $user['role_id'];
                $_SESSION['logged_in'] = true;
                
                return true;
            }
        }
    } catch (Exception $e) {
        error_log("Login error: " . $e->getMessage());
    }
    
    return false;
}

function handleGetRequest($path) {
    global $conn;
    
    if (preg_match('/\/api\/customers\/(\d+)/', $path, $matches)) {
        $customerId = $matches[1];
        
        $stmt = $conn->prepare("SELECT * FROM customers WHERE id = ?");
        $stmt->bind_param('i', $customerId);
        $stmt->execute();
        $result = $stmt->get_result();
        
        if ($customer = $result->fetch_assoc()) {
            echo json_encode($customer);
        } else {
            http_response_code(404);
            echo json_encode(['error' => 'Customer not found']);
        }
    }
}
